
## 实现add(1,2)(3)这种效果

```
    function add(...a) {
      let sum = 0
      doAdd(...a)
      function doAdd(...a) {
        for (const i of a) {
          sum += i
        }
      }
      function addMore(...b) {
        doAdd(...b)
        return addMore;
      };

      addMore.toString = function () {
        return sum;
      };

      return addMore;
    };
    console.log(add(1, 2, 3)(2, 1)(33))

```
